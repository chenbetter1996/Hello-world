求无序序列前k大的数
算法 数据结构
算法&数据结构
发布日期:   2019-07-03
文章字数:   302
阅读时长:   1 分

考察的是排序算法，最直接的是全部排一遍。优化可以先考虑冒泡次数，然后是快排的变形。以及堆排序

package xyz.cglzwz.demo;

/**
 * 查找前k大
 * @author Mechrevo
 *
 */

public class KSort {

    public static void main(String[] args) {
        int[] a = new int[] {1, 4, 6, 7, 8, 9, 5, 6, 11};
        KSort instance = new KSort();
        //instance.solution1(a, 4);
        instance.solution2(a, 0, a.length-1, 5);
        instance.print(a);


    }

    /**
     * 冒泡k次,  O(k * n)
     * @param a
     * @param k
     */
    public void solution1(int[] a, int k) {
        for (int i = 0; i < k; ++i) {
            for (int j = a.length - 1; j > i; j--) {
                if (a[j-1] < a[j]) {
                    swap(a, j-1, j);
                }
            }
        }
    }

    /**
     * 快排思想，接近O(n)
     * @param a
     * @param k
     */
    public void solution2(int[] a, int left, int right, int k) {
        if (left < right) {
            int mid = partition(a, left, right);
            if (mid == k || k - mid == 1)
                return;
            else if (mid > k) {
                solution2(a, left, mid - 1, k);
            } else {
                solution2(a, mid + 1, right, k - mid);
            }
        }

    }

    public int partition(int[] a, int left, int right) {
        int pivot = a[left];
        int temp = left;
        while (left < right) {
            while (left < right && a[right] <= pivot)
                right--;
            while (left < right && a[left] >= pivot)
                left++;
            if (left < right) {
                swap(a, left, right);
            }
        }
        // 轴枢
        a[temp] = a[left];
        a[left] = pivot;
        return left;
    }


    public void swap(int[] a, int x, int y) {
        a[y] = a[x] ^ a[y];
        a[x] = a[x] ^ a[y];
        a[y] = a[x] ^ a[y];
    }

    public void print(int[] a) {
        for (int x : a) {
            System.out.print(x + " ");
        }
        System.out.println();
    }

}
