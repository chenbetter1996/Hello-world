        this.val = val;
    }

1. 使用栈

/**
 * 使用栈进行反转
 *
 * @param head
 * @return
 */
public Node reverseByStack(Node head) {
    if (head == null || head.next == null)
        return head;

    Stack<Node> stack = new Stack<>();

    // 压入栈中
    while (head != null) {
        stack.push(head);
        head = head.next;
    }

    // 出栈
    Node ans = stack.pop();
    Node cur = ans;

    while (!stack.isEmpty()) {
        cur.next = stack.pop();
        cur = cur.next;
    }
    // 不加死循环
    cur.next = null;

    return ans;
}

很容易来理解，注意的是出栈的操作，初战结束的最后一个元素1会指向之前的2，所有需要 cur.next = null。

2. 递归法

/**
 * 递归反转，类似栈
 *
 * @param head
 * @return
 */
public Node reverseByRecursion(Node head) {
    if (head == null || head.next == null)
        return head;

    Node newNode = reverseByRecursion(head.next);

    head.next.next = head;
    head.next = null;

    return newNode;
}

其实也是使用栈的一种形式，递归本身就是内部使用了栈。后面两句是以递归最小为替换，如最后为 D -> E -> null 反转为 E -> D -> null。

而多于两个也是如此，如 C -> (E -> D -> null)，这里的head是C，head.next是D，head.next.next = head就是保证了 D -> C，后续C -> null，返回E即可。

3. 三指针法

public Node reverseByLoop(Node head) {
    if (head == null || head.next == null) {
        return head;
    }

    Node preNode = null;
    Node nextNode = null;

    while(head != null) {
        // 保存为下次迭代准备
        nextNode = head.next;
        // 唯一替换的语句
        head.next = preNode;
        // 下一次迭代
        preNode = head;
        head = nextNode;
    }

    return preNode;
}

每次遍历的是当前节点，如果当前节点为空，就返回上一个节点即可。
对head操作的唯一语句是head.next = preNode; 就是把它和前一个替换，并不涉及后一个的操作。
